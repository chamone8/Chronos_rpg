# Instalar dependências
flutter pub get

# Rodar o projeto
flutter run

# Limpar build
flutter clean

# Conseguir dependências novamente
flutter pub get

# Analisar código
flutter analyze

flutter run -d <device_id>

# Para listar dispositivos disponíveis:
flutter devices
